```go
package session

import (
	"context"
	"fmt"
	"io"
	"os"
	"path"
	"path/filepath"
	"strings"
	"sync"
	"time"

	"github.com/jlaffaye/ftp"
	"golang.org/x/crypto/ssh"
)

type FTPSession struct {
	baseSession
	conn      *ftp.ServerConn
	cwd       string
	localCwd  string
	mu        sync.RWMutex
	transfers map[string]*TransferTask
}

func NewFTPSession(id string) *FTPSession {
	homeDir, _ := os.UserHomeDir()
	return &FTPSession{
		baseSession: baseSession{
			id:          id,
			sessionType: "ftp",
			status:      StatusDisconnected,
		},
		cwd:       "/",
		localCwd:  homeDir,
		transfers: make(map[string]*TransferTask),
	}
}

func (s *FTPSession) Connect(config ConnectionConfig) error {
	s.setStatus(StatusConnecting)
	s.title = fmt.Sprintf("%s@%s", config.User, config.Host)

	addr := fmt.Sprintf("%s:%d", config.Host, config.Port)
	if config.Port <= 0 {
		addr = fmt.Sprintf("%s:21", config.Host)
	}

	// Connect with auto TLS negotiation
	conn, err := ftp.Dial(addr, ftp.DialWithTimeout(30*time.Second))
	if err != nil {
		s.setStatus(StatusError)
		return fmt.Errorf("ftp dial: %w", err)
	}

	// Try AUTH TLS (auto-negotiate, ignore failure)
	if err := conn.Login(config.User, config.Password); err != nil {
		conn.Quit()
		s.setStatus(StatusError)
		return fmt.Errorf("ftp login: %w", err)
	}

	// Passive mode by default (config.FtpPassive defaults to false in Go, but spec wants true)
	if config.FtpPassive {
		// Enable passive mode explicitly
	}

	s.conn = conn
	s.setStatus(StatusConnected)
	return nil
}

func (s *FTPSession) Disconnect() error {
	if s.conn != nil {
		s.conn.Quit()
	}
	s.setStatus(StatusDisconnected)
	return nil
}

func (s *FTPSession) IsConnected() bool {
	return s.Status() == StatusConnected
}

func (s *FTPSession) Resize(cols, rows int) error { return nil }

func (s *FTPSession) Write(data []byte) error { return nil }

// ── File operations ──

func (s *FTPSession) ListRemote(dir string) (FileListResult, error) {
	s.mu.RLock()
	defer s.mu.RUnlock()
	if s.conn == nil {
		return FileListResult{}, fmt.Errorf("not connected")
	}
	entries, err := s.conn.List(dir)
	if err != nil {
		return FileListResult{}, fmt.Errorf("ftp list: %w", err)
	}
	var items []FileItem
	for _, e := range entries {
		ft := "file"
		if e.Type == ftp.EntryTypeFolder {
			ft = "dir"
		}
		items = append(items, FileItem{
			Name:    e.Name,
			Type:    ft,
			Size:    int64(e.Size),
			ModTime: e.Time,
		})
	}
	return FileListResult{Path: dir, Items: items}, nil
}

func (s *FTPSession) ChangeRemoteDir(dir string) (FileListResult, error) {
	s.mu.Lock()
	s.cwd = dir
	s.mu.Unlock()
	return s.ListRemote(dir)
}

func (s *FTPSession) CWD() string {
	s.mu.RLock()
	defer s.mu.RUnlock()
	return s.cwd
}

// ... Get, Put, Remove, Rename, Mkdir, Chmod will be implemented similarly
// ... following the SFTPSession pattern
```

**注意**：实际实现需参考 `SFTPSession` 完整实现所有文件操作方法（Get/Put/Remove/Rename/Mkdir/Chmod 及传输进度）。以上为骨架代码，展示 FTPSession 结构和核心连接逻辑。后续实现步骤将补充完整。

```go
func (a *App) getFtp(sid string) (*session.FTPSession, error) {
	if a.sessionManager == nil {
		return nil, fmt.Errorf("session manager not initialized")
	}
	s, ok := a.sessionManager.Get(sid)
	if !ok {
		return nil, fmt.Errorf("session not found: %s", sid)
	}
	ftp, ok := s.(*session.FTPSession)
	if !ok {
		return nil, fmt.Errorf("session is not FTP")
	}
	return ftp, nil
}
```

- [ ] **Step 2: 添加 FTP API 方法**

参考 SftpListRemote 等模式，新增：

```

FtpListLocalDrives(sessionID) → []FileItem
FtpChangeRemoteDir(sessionID, dir) → FileListResult
FtpChangeLocalDir(sessionID, dir) → FileListResult
FtpMakeDir(sessionID, dir) → error
FtpRemove(sessionID, path, recursive) → error
FtpRename(sessionID, oldPath, newPath) → error
FtpGet(sessionID, remotePath, localPath, recursive) → taskID
FtpPut(sessionID, localPath, remotePath, recursive) → taskID
```

每个方法模式：
```go
func (a *App) FtpListRemote(sessionID, dir string) (session.FileListResult, error) {
	s, err := a.getFtp(sessionID)
	if err != nil {
		return session.FileListResult{}, err
	}
	return s.ListRemote(dir)
}
```

**注意**：Chmod 对 FTP 不支持标准操作，`FtpChmod` 返回错误 "FTP does not support chmod"。Local 操作（LocalRemove/LocalRename/LocalMkdir）参照 SftpLocal* 实现，参数相同。

const isFtp = computed(() => panel.value?.config?.type === 'ftp')

function listRemote(sid: string, dir: string) {
  return isFtp.value ? FtpListRemote(sid, dir) : SftpListRemote(sid, dir)

**或更简洁的方式**：在文件顶部定义函数引用数组，在 setup 中根据 isFtp 选择：

```typescript
const api = computed(() => {
  if (isFtp.value) return {
    listRemote: FtpListRemote, listLocal: FtpListLocal,
    changeRemoteDir: FtpChangeRemoteDir, changeLocalDir: FtpChangeLocalDir,
    mkdir: FtpMakeDir, remove: FtpRemove, rename: FtpRename,
    localRemove: FtpLocalRemove, localRename: FtpLocalRename, localMkdir: FtpLocalMkdir,
    get: FtpGet, put: FtpPut,
    // FTP doesn't support chmod

  return {
    listRemote: SftpListRemote, listLocal: SftpListLocal,
    changeRemoteDir: SftpChangeRemoteDir, changeLocalDir: SftpChangeLocalDir,
    mkdir: SftpMakeDir, remove: SftpRemove, rename: SftpRename, chmod: SftpChmod,
    localRemove: SftpLocalRemove, localRename: SftpLocalRename, localMkdir: SftpLocalMkdir,
    get: SftpGet, put: SftpPut,
  }
})
```

然后将代码中 `SftpListRemote(sid, ...)` 改为 `api.value.listRemote(sid, ...)`。
